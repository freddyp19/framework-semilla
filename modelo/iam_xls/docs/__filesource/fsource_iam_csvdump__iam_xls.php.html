  80.      function WriteSQLDump($query, $db, $user='root', $pass='', $host='localhost')
    81. 

  81.      {
    82. 

  82.           $xls_line  = 0;
    83. 

  83.           $col = 0;
    84. 

  84.  
    85. 

  85.           $link = $this->_db_connect($db, $user, $pass, $host);
    86. 

  86.           if($link)
    87. 

  87.           {
    88. 

  88.                $result= @mysql_query($query, $link);
    89. 

  89.                if(mysql_error()!="")
    90. 

  90.                {
    91. 

  91.                     $this->WriteValue(1, 0, "Ha ocurrido un error al consultar la base de datos: ".mysql_error());
    92. 

  92.                     $this->OutputFile();
    93.

    94. 

  94.                }
    95. 

  95.  
    96. 

  96.                $lines  = @mysql_num_rows($result);

    98. 

  98.  
    99. 

  99.                for($e=0; $e < $colums; $e++)

  100.                     $this->WriteValue(0, $e, trim(ucwords(str_replace("_"," ",mysql_field_name($result, $e)))));
    101. 

  101.  
    102. 

  102.                for($col = 0; $col < $colums; $col++)
    103. 

  103.                {
    104. 

  104.                    $col_name  = mysql_field_name($result, $col);
    105. 

  105.  
    106. 

  106.                     for($i=0; $i < $lines; $i++)
    107. 

  107.                    {
    108. 

  108.                               $CellValue = mysql_result($result, $i, $col_name);
    109. 

  109.  
    110. 

  110.                              $xls_line = ($i + 1);
    111. 

  111.  
    112. 

  112.                              $this->WriteValue($xls_line, $col, $CellValue);
    113. 

  113.                    }
    114. 

  114.                }
    115. 

  115.           }
    116. 

  116.           else
    117. 

  117.           {
    118. 

  118.                $this->WriteValue(1, 0, "Ha ocurrido un error al conectarse a la base de datos: ".mysql_error());
    119. 

  119.           }
    120. 

  120.           $this->OutputFile();
    121. 

  121.      }
